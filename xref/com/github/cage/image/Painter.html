108 	 * @param quality
109 	 *            captcha image quality, default {@link Quality#MAX}, should use
110 	 *            max it does not have measurable speed penalty on modern jvm-s

112 	 * @param ripple
113 	 *            waving effect should be used, default true, disabling this

116 	 *            should the image be blurred, default true, disabling this
117 	 *            helps performance

128 	public Painter(int width, int height, Color bGround, Quality quality,
129 			boolean ripple, boolean blur, boolean outline, boolean rotate,
130 			Random rnd) {
131 		super();
132 		this.width = width;
133 		this.height = height;
134 		this.background = bGround != null ? bGround : Color.WHITE;
135 		this.quality = quality != null ? quality : Quality.MAX;
136 		this.rippleEnabled = ripple;
137 		this.blurEnabled = blur;
138 		this.outlineEnabled = outline;
139 		this.rotateEnabled = rotate;
140 		this.rnd = rnd != null ? rnd : new Random();
141 	}

143 	/**
144 	 * Generates a new captcha image.
145 	 * 
146 	 * @param font
147 	 *            will be used for text, not null
148 	 * @param fGround
149 	 *            will be used for text, not null
150 	 * @param text
151 	 *            this will be rendered on the image, not null, not 0 length
152 	 * @return the generated image
153 	 */
154 	public BufferedImage draw(Font font, Color fGround, String text) {
155 		if (font == null)
156 			throw new IllegalArgumentException("Font can not be null.");
157 		if (fGround == null)
158 			throw new IllegalArgumentException(
159 					"Foreground color can not be null.");
160 		if (text == null || text.length() < 1)
161 			throw new IllegalArgumentException("No text given.");
162 
163 		BufferedImage img = createImage();
164 
165 		Graphics g = img.getGraphics();
166 		try {
167 			Graphics2D g2 = configureGraphics(g, font, fGround);
168 
169 			draw(g2, text);
170 		} finally {
171 			g.dispose();
172 		}
173 
174 		img = postProcess(img);
175 
176 		return img;
177 	}

185 		return new BufferedImage(width, height, BufferedImage.TYPE_3BYTE_BGR);
186 	}

188 	/**
189 	 * Configures graphics object before drawing text on it.
190 	 * 
191 	 * @param g
192 	 *            to be configured must be an instance of {@link Graphics2D},
193 	 *            not null
194 	 * @param font
195 	 *            to be used for the text, not null
196 	 * @param fGround
197 	 *            to be used for the text, not null
198 	 * 
199 	 * @return g casted to {@link Graphics2D} or throws exception if g is not
200 	 *         instance of {@link Graphics2D}.
201 	 */
202 	protected Graphics2D configureGraphics(Graphics g, Font font, Color fGround) {
203 		if (!(g instanceof Graphics2D))
204 			throw new IllegalStateException("Graphics (" + g
205 					+ ") that is not an instance of Graphics2D.");
206 		Graphics2D g2 = (Graphics2D) g;
207 
208 		configureGraphicsQuality(g2);
209 
210 		g2.setColor(fGround);
211 		g2.setBackground(background);
212 		g2.setFont(font);
213 
214 		g2.clearRect(0, 0, width, height);
215 
216 		return g2;
217 	}

219 	/**
220 	 * Sets quality related hints based on the quality field of this object.
221 	 * 
222 	 * @param g2
223 	 *            to be configured, not null
224 	 */
225 	protected void configureGraphicsQuality(Graphics2D g2) {
226 		switch (quality) {
227 		case MAX:
228 			g2.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
229 					RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
230 			g2.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
231 					RenderingHints.VALUE_FRACTIONALMETRICS_ON);
232 			g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
233 					RenderingHints.VALUE_ANTIALIAS_ON);
234 			g2.setRenderingHint(RenderingHints.KEY_COLOR_RENDERING,
235 					RenderingHints.VALUE_COLOR_RENDER_QUALITY);
236 			g2.setRenderingHint(RenderingHints.KEY_DITHERING,
237 					RenderingHints.VALUE_DITHER_ENABLE);
238 			g2.setRenderingHint(RenderingHints.KEY_INTERPOLATION,
239 					RenderingHints.VALUE_INTERPOLATION_BICUBIC);
240 			g2.setRenderingHint(RenderingHints.KEY_ALPHA_INTERPOLATION,
241 					RenderingHints.VALUE_ALPHA_INTERPOLATION_QUALITY);
242 			g2.setRenderingHint(RenderingHints.KEY_RENDERING,
243 					RenderingHints.VALUE_RENDER_QUALITY);
244 			break;
245 		case MIN:
246 			g2.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
247 					RenderingHints.VALUE_TEXT_ANTIALIAS_OFF);
248 			g2.setRenderingHint(RenderingHints.KEY_FRACTIONALMETRICS,
249 					RenderingHints.VALUE_FRACTIONALMETRICS_OFF);
250 			g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
251 					RenderingHints.VALUE_ANTIALIAS_OFF);
252 			g2.setRenderingHint(RenderingHints.KEY_COLOR_RENDERING,
253 					RenderingHints.VALUE_COLOR_RENDER_SPEED);
254 			g2.setRenderingHint(RenderingHints.KEY_DITHERING,
255 					RenderingHints.VALUE_DITHER_DISABLE);
256 			g2.setRenderingHint(RenderingHints.KEY_INTERPOLATION,
257 					RenderingHints.VALUE_INTERPOLATION_NEAREST_NEIGHBOR);
258 			g2.setRenderingHint(RenderingHints.KEY_ALPHA_INTERPOLATION,
259 					RenderingHints.VALUE_ALPHA_INTERPOLATION_SPEED);
260 			g2.setRenderingHint(RenderingHints.KEY_RENDERING,
261 					RenderingHints.VALUE_RENDER_SPEED);
262 			break;
263 		}
264 	}

266 	/**
267 	 * Does some of the text transformation (calls
268 	 * {@link #transform(Graphics2D, String, GlyphVector)}), scales, transforms
269 	 * and draws the result (also the outline if needed).
270 	 * 
271 	 * @param g
272 	 *            to be drawn upon
273 	 * @param text
274 	 *            to be drawn
275 	 */
276 	protected void draw(Graphics2D g, String text) {
277 		GlyphVector vector = g.getFont().createGlyphVector(
278 				g.getFontRenderContext(), text);
279 
280 		transform(g, text, vector);
281 
282 		Rectangle bounds = vector.getPixelBounds(null, 0, height);
283 		float bw = (float) bounds.getWidth();
284 		float bh = (float) bounds.getHeight();
285 
286 		// transform + scale text to better fit the image
287 		float wr = width / bw
288 				* (rnd.nextFloat() / 20 + (outlineEnabled ? 0.89f : 0.92f));
289 		float hr = height / bh
290 				* (rnd.nextFloat() / 20 + (outlineEnabled ? 0.68f : 0.75f));
291 		g.translate((width - bw * wr) / 2, (height - bh * hr) / 2);
292 		g.scale(wr, hr);
293 
294 		float bx = (float) bounds.getX();
295 		float by = (float) bounds.getY();
296 		// draw outline if needed
297 		if (outlineEnabled)
298 			g.draw(vector.getOutline(Math.signum(rnd.nextFloat() - 0.5f) * 1
299 					* width / 200 - bx, Math.signum(rnd.nextFloat() - 0.5f) * 1
300 					* height / 70 + height - by));
301 		g.drawGlyphVector(vector, -bx, height - by);
302 	}

304 	/**
305 	 * Does some of the text transformation (like rotation and symbol crowding).
306 	 * 
307 	 * @param g
308 	 *            to be drawn upon
309 	 * @param text
310 	 *            to be drawn
311 	 * @param v
312 	 *            graphical representation of text, to be transformed
313 	 */
314 	protected void transform(Graphics2D g, String text, GlyphVector v) {
315 		int glyphNum = v.getNumGlyphs();
316 
317 		Point2D prePos = null;
318 		Rectangle2D preBounds = null;
319 
320 		double rotateCur = (rnd.nextDouble() - 0.5) * Math.PI / 8;
321 		double rotateStep = Math.signum(rotateCur)
322 				* (rnd.nextDouble() * 3 * Math.PI / 8 / glyphNum);
323 
324 		for (int fi = 0; fi < glyphNum; fi++) {
325 			if (rotateEnabled) {
326 				AffineTransform tr = AffineTransform
327 						.getRotateInstance(rotateCur);
328 				if (rnd.nextDouble() < 0.25)
329 					rotateStep *= -1;
330 				rotateCur += rotateStep;
331 				v.setGlyphTransform(fi, tr);
332 			}
333 			Point2D pos = v.getGlyphPosition(fi);
334 			Rectangle2D bounds = v.getGlyphVisualBounds(fi).getBounds2D();
335 			Point2D newPos;
336 			if (prePos == null)
337 				newPos = new Point2D.Double(pos.getX() - bounds.getX(),
338 						pos.getY());
339 			else
340 				newPos = new Point2D.Double(
341 						preBounds.getMaxX()
342 								+ pos.getX()
343 								- bounds.getX()
344 								- Math.min(preBounds.getWidth(),
345 										bounds.getWidth())
346 								* (rnd.nextDouble() / 20 + (rotateEnabled ? 0.27
347 										: 0.20)), pos.getY());
348 			v.setGlyphPosition(fi, newPos);
349 			prePos = newPos;
350 			preBounds = v.getGlyphVisualBounds(fi).getBounds2D();
351 		}
352 	}

354 	/**
355 	 * Does some post processing on the generated image if needed. Like rippling
356 	 * (waving) and blurring.
357 	 * 
358 	 * @param img
359 	 *            to be post prosessed.
360 	 * @return the finished image, maybe the same as the input
361 	 */
362 	protected BufferedImage postProcess(BufferedImage img) {
363 		if (rippleEnabled) {
364 			Rippler.AxisConfig vertical = new Rippler.AxisConfig(
365 					rnd.nextDouble() * 2 * Math.PI, (1 + rnd.nextDouble() * 2)
366 							* Math.PI, img.getHeight() / 10.0);
367 			Rippler.AxisConfig horizontal = new Rippler.AxisConfig(
368 					rnd.nextDouble() * 2 * Math.PI, (2 + rnd.nextDouble() * 2)
369 							* Math.PI, img.getWidth() / 100.0);
370 			Rippler op = new Rippler(vertical, horizontal);
371 
372 			img = op.filter(img, createImage());
373 		}
374 		if (blurEnabled) {
375 			float[] blurArray = new float[9];
376 			fillBlurArray(blurArray);
377 			ConvolveOp op = new ConvolveOp(new Kernel(3, 3, blurArray),
378 					ConvolveOp.EDGE_NO_OP, null);
379 
380 			img = op.filter(img, createImage());
381 		}
382 		return img;
383 	}

385 	/**
386 	 * Generates a random probability distribution. Used by blurring.
387 	 * 
388 	 * @param array
389 	 *            filled with random values. The values in array sum up to 1.
390 	 */
391 	protected void fillBlurArray(float[] array) {
392 		float sum = 0;
393 		for (int fi = 0; fi < array.length; fi++)
394 			sum += array[fi] = rnd.nextFloat();
395 		for (int fi = 0; fi < array.length; fi++)
396 			array[fi] /= sum;
397 	}
