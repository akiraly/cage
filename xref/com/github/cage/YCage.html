16  package com.github.cage;
17  
18  import java.awt.Color;
19  import java.util.Locale;
20  import java.util.Random;
21  
22  import com.github.cage.image.ConstantColorGenerator;
23  import com.github.cage.image.Painter;
24  import com.github.cage.token.RandomCharacterGeneratorFactory;
25  import com.github.cage.token.RandomTokenGenerator;
26  
27  /**
28   * Creates and configures a {@link Cage} instance that can generate captcha
29   * images similar to Yahoo's. This is the "Y" template. Simply create an
30   * instance with <code>new YCage()</code> and you can generate images. See
31   * {@link Cage} for more info.
32   * 
33   * This class is thread safe.
34   * 
35   * @author akiraly
36   */
37  public class YCage extends com.github.cage.Cage {
38  	/**
39  	 * Height of CAPTCHA image.
40  	 */
41  	protected static final int HEIGHT = 80;
42  
43  	/**
44  	 * Width of CAPTCHA image.
45  	 */
46  	protected static final int WIDTH = 290;
47  
48  	/**
49  	 * Character set supplied to the {@link RandomTokenGenerator} used by this
50  	 * template.
51  	 */
52  	protected static final char[] TOKEN_DEFAULT_CHARACTER_SET = (new String(
53  			RandomCharacterGeneratorFactory.DEFAULT_DEFAULT_CHARACTER_SET)
54  			.replaceAll("b|i|j|l|o", "")
55  			+ new String(
56  					RandomCharacterGeneratorFactory.DEFAULT_DEFAULT_CHARACTER_SET)
57  					.replaceAll("i|o", "").toUpperCase(Locale.ENGLISH) + new String(
58  			RandomCharacterGeneratorFactory.ARABIC_NUMERALS).replaceAll("0|1|9",
59  			"")).toCharArray();
60  
61  	/**
62  	 * Minimum length of token.
63  	 */
64  	protected static final int TOKEN_LEN_MIN = 6;
65  
66  	/**
67  	 * Maximum length of token is {@value #TOKEN_LEN_MIN} +
68  	 * {@value #TOKEN_LEN_DELTA}.
69  	 */
70  	protected static final int TOKEN_LEN_DELTA = 2;
71  
72  	/**
73  	 * Constructor.
74  	 */
75  	public YCage() {
76  		this(new Random());
77  	}
78  
79  	/**
80  	 * Constructor.
81  	 * 
82  	 * @param rnd
83  	 *            object used for random value generation. Not null.
84  	 */
85  	protected YCage(Random rnd) {
86  		super(new Painter(WIDTH, HEIGHT, null, null, new Painter.EffectConfig(
87  				true, true, true, false), rnd), null,
88  				new ConstantColorGenerator(Color.BLACK), null,
89  				Cage.DEFAULT_COMPRESS_RATIO, new RandomTokenGenerator(rnd,
90  						new RandomCharacterGeneratorFactory(
91  								TOKEN_DEFAULT_CHARACTER_SET, null, rnd),
92  						TOKEN_LEN_MIN, TOKEN_LEN_DELTA), rnd);
93  	}
94  }
