16  package com.github.cage;
17  
18  import java.awt.Color;
19  import java.util.Locale;
20  import java.util.Random;
21  
22  import com.github.cage.image.ConstantColorGenerator;
23  import com.github.cage.image.Painter;
24  import com.github.cage.token.RandomCharacterGeneratorFactory;
25  import com.github.cage.token.RandomTokenGenerator;
26  
27  /**
28   * Creates and configures a {@link Cage} instance that can generate captcha
29   * images similar to Yahoo's. This is the "Y" template. Simply create an
30   * instance with <code>new YCage()</code> and you can generate images. See
31   * {@link Cage} for more info.
32   * 
33   * This class is thread safe.
34   * 
35   * @author akiraly
36   */
37  public class YCage extends com.github.cage.Cage {
38  	/**
39  	 * Character set supplied to the {@link RandomTokenGenerator} used by this
40  	 * template.
41  	 */
42  	public static final char[] TOKEN_DEFAULT_CHARACTER_SET = (new String(
43  			RandomCharacterGeneratorFactory.DEFAULT_DEFAULT_CHARACTER_SET)
44  			.replaceAll("b|i|j|l|o", "")
45  			+ new String(
46  					RandomCharacterGeneratorFactory.DEFAULT_DEFAULT_CHARACTER_SET)
47  					.replaceAll("i|o", "").toUpperCase(Locale.ENGLISH) + new String(
48  			RandomCharacterGeneratorFactory.ARAB_NUMBERS).replaceAll("0|1|9",
49  			"")).toCharArray();
50  
51  	/**
52  	 * Constructor.
53  	 */
54  	public YCage() {
55  		this(new Random());
56  	}
57  
58  	/**
59  	 * Constructor.
60  	 * 
61  	 * @param rnd
62  	 *            object used for random value generation. Not null.
63  	 */
64  	protected YCage(Random rnd) {
65  		super(new Painter(290, 80, null, null, true, true, true, false, rnd),
66  				null, new ConstantColorGenerator(Color.BLACK), null,
67  				Cage.DEFAULT_COMPRESS_RATIO, new RandomTokenGenerator(rnd,
68  						new RandomCharacterGeneratorFactory(
69  								TOKEN_DEFAULT_CHARACTER_SET, null, rnd), 6, 2),
70  				rnd);
71  	}
72  }
